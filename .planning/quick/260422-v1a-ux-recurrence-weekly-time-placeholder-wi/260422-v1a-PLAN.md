**1.2. `client/core/storage.py::update_task` — клонирование:**

Добавить логику клонирования ВНУТРИ метода `update_task` (строки 138-159), до `self._save_locked()`:

```python
def update_task(self, task_id: str, **fields) -> bool:
    allowed = {"text", "day", "time_deadline", "done", "position", "recurrence"}  # + recurrence
    unknown = set(fields) - allowed
    if unknown:
        raise ValueError(f"Неизвестные поля: {unknown}")

    clone_after: Optional[Task] = None  # готовим клон до выхода из lock

    with self._lock:
        target = None
        for t in self._data["tasks"]:
            if t.get("id") == task_id:
                target = t
                break
        if target is None or target.get("deleted_at") is not None:
            return False

        prev_done = bool(target.get("done", False))
        new_done = fields.get("done", prev_done)

        for k, v in fields.items():
            target[k] = v
        target["updated_at"] = utcnow_iso()

        # Recurrence клон: done переходит False→True и recurrence == "weekly"
        if (
            not prev_done and new_done
            and target.get("recurrence") == "weekly"
            and target.get("day")
        ):
            try:
                from datetime import date as _date, timedelta as _td
                next_day = (_date.fromisoformat(target["day"]) + _td(days=7)).isoformat()
                clone_after = Task.new(
                    user_id=target.get("user_id", ""),
                    text=target.get("text", ""),
                    day=next_day,
                    time_deadline=target.get("time_deadline"),
                    position=int(target.get("position", 0)) + 1000,
                )
                clone_after.recurrence = "weekly"
            except (ValueError, TypeError):
                clone_after = None

        # recurrence НЕ попадает в pending_changes (TaskChange не знает про него) —
        # фильтруем вручную, чтобы не бросить ValueError на сервере.
        wire_fields = {k: v for k, v in fields.items() if k != "recurrence"}
        if wire_fields:
            change = TaskChange(op="update", task_id=task_id, **wire_fields)
            self._data["pending_changes"].append(change.to_dict())
        self._save_locked()

    # add_task берёт lock сам — вызываем ПОСЛЕ выхода из with (Lock не RLock)
    if clone_after is not None:
        self.add_task(clone_after)

    return True
```

Ключевые моменты:
- `allowed` расширен для `recurrence`
- `clone_after` готовится внутри lock, но `add_task()` вызывается СНАРУЖИ lock (threading.Lock не рекурсивна — D-12)
- `recurrence` фильтруется из `TaskChange` — сервер про него не знает, иначе 422
- Position клона = parent + 1000 (низкий приоритет в списке, владелец переставит если нужно)
- `clone.recurrence = "weekly"` — клон тоже повторяется (цепная генерация каждую неделю)

                    pass

                500, self._debounced_save_window_state

    except tk.TclError:
        pass

**4.3. Вспомогательный `_blend_hex` (скопировать из inline_edit_panel.py):**

```python
@staticmethod
def _blend_hex(a: str, b: str, t: float) -> str:
    """Линейный блендинг двух hex-цветов. t=0 → a, t=1 → b."""
    def _parse(h: str) -> tuple[int, int, int]:
        h = h.lstrip("#")
        return (int(h[0:2], 16), int(h[2:4], 16), int(h[4:6], 16))
    ar, ag, ab = _parse(a)
    br, bg, bb = _parse(b)
    r = int(ar + (br - ar) * t)
    g = int(ag + (bg - ag) * t)
    bl = int(ab + (bb - ab) * t)
    return f"#{r:02X}{g:02X}{bl:02X}"
```

```python
def show_at_overlay(self, overlay_x: int, overlay_y: int, overlay_size: int = 73) -> None:
    """Показать popup рядом с overlay. Clamp к границам экрана.

    Приоритет размещения:
      1. Справа от overlay (overlay_x + overlay_size + gap)
      2. Слева от overlay (если не влез справа)
      3. Центрирование по X относительно overlay (если не влез ни справа, ни слева)
      Y: по умолчанию = overlay_y; clamp к screen bounds.
    """
    if self._visible:
        self.hide()
        return

    try:
        sw = self._root.winfo_screenwidth()
        sh = self._root.winfo_screenheight()
    except tk.TclError:
        sw, sh = 1920, 1080

    # X positioning
    x = overlay_x + overlay_size + self.POPUP_GAP  # prefer right
    if x + self.POPUP_WIDTH + self.EDGE_MARGIN > sw:
        # Try left
        x = overlay_x - self.POPUP_WIDTH - self.POPUP_GAP
    if x < self.EDGE_MARGIN:
        # Fallback: center horizontally relative to overlay, clamped
        x = overlay_x + overlay_size // 2 - self.POPUP_WIDTH // 2
        x = max(self.EDGE_MARGIN, min(x, sw - self.POPUP_WIDTH - self.EDGE_MARGIN))

    # Y positioning
    y = overlay_y
    if y + self.POPUP_HEIGHT + self.EDGE_MARGIN > sh:
        y = sh - self.POPUP_HEIGHT - self.EDGE_MARGIN
    if y < self.EDGE_MARGIN:
        y = self.EDGE_MARGIN

    self._create_popup(x, y)
```

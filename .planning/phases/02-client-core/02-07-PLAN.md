    ```python
    """
    SyncManager — фоновая синхронизация (D-07, D-08, D-13, D-19..D-21).

    Алгоритм одного цикла:
        1. drain_pending_changes()  # атомарно изъять локальные изменения
        2. определить since = last_sync_at | None (None при stale > 5 мин)
        3. api_client.post_sync(since, drained)
        4. На успех:
             - storage.merge_from_server(server.changes, server_timestamp)
             - opportunistic cleanup_tombstones (если нечего push)
        5. На failure:
             - restore_pending_changes(drained)
             - если auth_expired — остановить thread (нужен Telegram-код)

    Wake-up: threading.Event. UI thread вызывает force_sync() → Event.set() →
    sync thread просыпается из wait(timeout=...).
    """
    from __future__ import annotations

    import logging
    import threading
    from datetime import datetime, timezone
    from typing import Optional

    from client.core import config
    from client.core.api_client import ApiResult, SyncApiClient
    from client.core.auth import AuthManager
    from client.core.storage import LocalStorage

    logger = logging.getLogger(__name__)


    class SyncManager:
        """Daemon thread оркестратор синхронизации."""

        def __init__(
            self,
            storage: LocalStorage,
            auth_manager: AuthManager,
            api_client: Optional[SyncApiClient] = None,
        ) -> None:
            self._storage = storage
            self._auth = auth_manager
            self._api_client = api_client or SyncApiClient(auth_manager)

            self._stop_event = threading.Event()
            self._wake_event = threading.Event()
            self._thread: Optional[threading.Thread] = None
            self._auth_expired: bool = False

        # ---- public API ----

        def is_running(self) -> bool:
            return self._thread is not None and self._thread.is_alive()

        def start(self) -> None:
            """Запустить sync thread (idempotent)."""
            if self.is_running():
                logger.debug("SyncManager.start: уже запущен")
                return
            self._stop_event.clear()
            self._wake_event.clear()
            self._auth_expired = False
            self._thread = threading.Thread(
                target=self._sync_loop, daemon=True, name="PlannerSync"
            )
            self._thread.start()
            logger.info("SyncManager started")

        def stop(self, timeout: float = 5.0) -> None:
            """Остановить sync thread (graceful)."""
            self._stop_event.set()
            self._wake_event.set()  # разбудить если ждёт
            if self._thread is not None:
                self._thread.join(timeout=timeout)
                if self._thread.is_alive():
                    logger.warning("SyncManager.stop: thread не завершился за %.1fs", timeout)
            self._thread = None
            logger.info("SyncManager stopped")

        def force_sync(self) -> None:
            """D-08 / D-21: разбудить sync thread немедленно (immediate push)."""
            if not self.is_running():
                logger.debug("force_sync: SyncManager не запущен — игнорируем")
                return
            self._wake_event.set()

        # ---- internal: thread loop ----

        def _sync_loop(self) -> None:
            """Основной цикл daemon thread."""
            while not self._stop_event.is_set():
                try:
                    self._attempt_sync()
                except Exception as exc:  # noqa: BLE001
                    # Защитная сетка — не позволяем потоку умереть по неожиданной ошибке
                    logger.exception("Unexpected sync error: %s", exc)

                if self._auth_expired:
                    logger.warning("Sync loop exiting — auth expired")
                    break

                # Если api_client в backoff — ждём backoff; иначе штатный интервал
                if self._api_client.consecutive_errors > 0:
                    wait_time = self._api_client.current_backoff
                else:
                    wait_time = config.SYNC_INTERVAL_SECONDS

                self._wake_event.wait(timeout=wait_time)
                self._wake_event.clear()

        # ---- internal: one sync attempt ----

        def _attempt_sync(self) -> ApiResult:
            """
            Один цикл sync. Возвращает ApiResult для тестов и логирования.
            Никогда не raise — все ошибки обрабатываются и логируются.
            """
            # Не пытаемся sync если нет access_token
            if self._auth.get_access_token() is None:
                logger.debug("_attempt_sync: нет access_token, пропускаем")
                return ApiResult(ok=False, status=0, error_kind="auth", message="no token")

            last_sync_at = self._storage.get_meta("last_sync_at")
            is_stale = self._is_stale(last_sync_at)

            # D-20: drain pending ПЕРЕД full resync (чтобы локальные изменения не потерялись)
            drained = self._storage.drain_pending_changes()

            # Skip-условие: ничего push'ить, не stale, last_sync был → нет смысла дёргать сервер
            if not drained and not is_stale and last_sync_at is not None:
                # Возвращаем "успешный noop" — backoff не растёт, цикл ждёт штатные 30s
                return ApiResult.success({
                    "changes": [],
                    "server_timestamp": last_sync_at,
                })

            since = None if is_stale else last_sync_at
            logger.debug("Sync attempt: drained=%d, since=%s, stale=%s",
                         len(drained), since, is_stale)

            result = self._api_client.post_sync(since=since, changes=drained)

            if result.ok:
                payload = result.payload or {}
                server_changes = payload.get("changes", []) or []
                server_timestamp = payload.get("server_timestamp", "")
                stats = self._storage.merge_from_server(server_changes, server_timestamp)
                self._storage.commit_drained(drained)
                logger.info(
                    "Sync OK: pushed=%d, applied=%d, conflicts=%d, tombstones=%d",
                    len(drained), stats["applied"], stats["conflicts"], stats["tombstones_received"],
                )
                # Opportunistic cleanup (D-23, D-24): если очередь пустая — попробуем удалить старые tombstones
                if self._storage.pending_count() == 0:
                    self._storage.cleanup_tombstones()
                return result

            # Failure path: вернуть pending в очередь
            if drained:
                self._storage.restore_pending_changes(drained)

            if result.error_kind == "auth_expired":
                self._auth_expired = True
                logger.critical("Sync auth_expired — требуется повторный логин")
            elif result.error_kind == "client":
                # 4xx (не 401): bug на клиенте — НЕ ретраим бесконечно
                logger.error("Sync client error %d: %s", result.status, result.message)
                # Не auth_expired, но тоже остановим — иначе infinite loop
                self._auth_expired = True
            else:
                logger.warning(
                    "Sync failed (%s, status=%d): %s — retry через %.1fs",
                    result.error_kind, result.status, result.message,
                    result.retry_after or config.SYNC_INTERVAL_SECONDS,
                )
            return result

        @staticmethod
        def _is_stale(last_sync_at: Optional[str]) -> bool:
            """D-19: stale если last_sync_at старше STALE_THRESHOLD_SECONDS."""
            if last_sync_at is None:
                return True
            try:
                last = datetime.fromisoformat(last_sync_at.replace("Z", "+00:00"))
            except (ValueError, AttributeError, TypeError):
                return True
            try:
                delta = (datetime.now(timezone.utc) - last).total_seconds()
            except (TypeError, OverflowError):
                return True
            return delta > config.STALE_THRESHOLD_SECONDS
    ```

    class FakeServer:
        """Stateful mock сервера. Хранит tasks, возвращает delta по since."""

        def __init__(self) -> None:
            self.tasks: dict[str, dict] = {}
            self.online: bool = True

        def now(self) -> str:
            return datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")

        def handle_sync(self, request, _ctx):
            if not self.online:
                # Симуляция offline — raise ConnectionError эмулирует requests-mock через exc
                # но в matcher это не сработает; используем 503
                _ctx.status_code = 503
                return {"error": {"code": "OFFLINE", "message": "server down"}}

            body = request.json()
            since = body.get("since")
            changes = body.get("changes", []) or []

            # Применить changes к нашему "БД"
            for ch in changes:
                op = ch["op"]
                tid = ch["task_id"]
                if op == "create" or op == "update":
                    if tid not in self.tasks:
                        self.tasks[tid] = {
                            "task_id": tid,
                            "text": "",
                            "day": "",
                            "time_deadline": None,
                            "done": False,
                            "position": 0,
                            "created_at": self.now(),
                            "updated_at": self.now(),
                            "deleted_at": None,
                        }
                    for k in ("text", "day", "time_deadline", "done", "position"):
                        if k in ch:
                            self.tasks[tid][k] = ch[k]
                    self.tasks[tid]["updated_at"] = self.now()
                elif op == "delete":
                    if tid in self.tasks:
                        self.tasks[tid]["deleted_at"] = self.now()
                        self.tasks[tid]["updated_at"] = self.now()

            # Delta: tasks с updated_at > since (если since задан)
            if since:
                returned = [t for t in self.tasks.values() if t["updated_at"] > since]
            else:
                returned = list(self.tasks.values())  # full resync

            return {
                "server_timestamp": self.now(),
                "changes": returned,
            }

    def test_offline_50_tasks_no_loss_after_reconnect(authed_setup):
        """SC-2: 50 offline tasks → reconnect → все 50 на сервере без потерь."""
        ctx = authed_setup
        storage, sync, server = ctx["storage"], ctx["sync"], ctx["server"]

        # Сервер "выключен"
        server.online = False
        task_ids = []
        for i in range(50):
            t = Task.new(user_id="user-1", text=f"task {i}", day="2026-04-14",
                         position=i)
            storage.add_task(t)
            task_ids.append(t.id)

        # Несколько неуспешных sync attempts (сервер 503)
        for _ in range(3):
            result = sync._attempt_sync()
            assert result.ok is False

        # Все 50 changes (по 1 CREATE на каждую) должны быть в pending
        # На самом деле может быть меньше — если drain+restore работает корректно
        assert storage.pending_count() >= 50
        assert len(server.tasks) == 0  # на сервере ничего

        # Сервер "ожил"
        server.online = True
        # Reset backoff чтобы _attempt_sync не игнорировал
        ctx["api_client"].reset_backoff()
        # Один sync — все changes отправлены

        assert result.ok is True
        # Все 50 на сервере
        assert len(server.tasks) == 50
        # Уникальные ids — no duplicates

    def test_server_wins_on_conflict(authed_setup):

        local = storage.get_task(t.id)

    def test_force_sync_in_running_thread(authed_setup):

    def test_concurrent_ui_writes_during_sync(authed_setup):
        """
        SYNC-04 в реальном flow: UI thread добавляет changes пока SyncManager
        одновременно делает _attempt_sync. Lock защищает state.
        """
        ctx = authed_setup
        storage, sync, server = ctx["storage"], ctx["sync"], ctx["server"]

        N_ADDS = 30
        added_ids: list[str] = []
        added_lock = threading.Lock()

        def ui_writer():
            for i in range(N_ADDS):
                t = Task.new(user_id="user-1", text=f"concurrent {i}",
                             day="2026-04-14")
                storage.add_task(t)
                with added_lock:
                    added_ids.append(t.id)
                time.sleep(0.005)  # имитация UI-events

        sync.start()
        try:
            writer = threading.Thread(target=ui_writer)
            writer.start()
            writer.join(timeout=10)
            # Дать sync время доработать
            time.sleep(1.0)
            sync.force_sync()
            time.sleep(2.0)
            # Все добавленные id должны быть на сервере
            missing = set(added_ids) - set(server.tasks.keys())
            assert not missing, f"Потеряны: {missing}"
        finally:
            sync.stop(timeout=2)

#!/usr/bin/env bash
# Bootstrap VPS for Личный Еженедельник.
# РАЗ запускается при первом deploy. Создаёт user, каталоги, права.
#
# Usage (на VPS как root):
#   sudo bash bootstrap-vps.sh
#
# Безопасно запускать повторно — все операции идемпотентны.
set -euo pipefail

echo "==> Создание user 'planner' (system, без shell)..."
if id planner &>/dev/null; then
    echo "  user planner уже существует — пропускаем"
else
    useradd -r -s /bin/false planner
fi

echo "==> Создание каталогов..."
mkdir -p /opt/planner
mkdir -p /var/lib/planner
mkdir -p /etc/planner

# Права
chown planner:planner /var/lib/planner
chmod 750 /var/lib/planner

# /etc/planner — root владеет, planner читает
chown root:planner /etc/planner
chmod 750 /etc/planner

# /opt/planner — planner владеет (git checkout + venv)
chown planner:planner /opt/planner
chmod 755 /opt/planner

echo "==> Cyrillic path sanity check..."
# PITFALLS.md: проверить что пути не содержат кириллицы (VPS на ASCII OK, но paranoid check)
for p in /opt/planner /var/lib/planner /etc/planner; do
    if echo "$p" | grep -P '[^\x00-\x7F]' > /dev/null; then
        echo "  ERROR: $p содержит не-ASCII символы — откажись от такого path"
        exit 1
    fi
done
echo "  все paths ASCII-safe"

echo "==> /etc/planner/planner.env..."
if [ ! -f /etc/planner/planner.env ]; then
    echo "  !!! /etc/planner/planner.env НЕ СУЩЕСТВУЕТ !!!"
    echo "  Создай его руками (runbook deploy/README.md):"
    echo "    sudo cp /opt/planner/server/.env.example /etc/planner/planner.env"
    echo "    sudo chmod 600 /etc/planner/planner.env"
    echo "    sudo chown root:planner /etc/planner/planner.env"
    echo "    sudo nano /etc/planner/planner.env  # заполни РЕАЛЬНЫЕ значения"
    echo ""
    echo "  БЕЗ этого сервис не стартует."
else
    echo "  /etc/planner/planner.env есть — проверяю права"
    chmod 600 /etc/planner/planner.env
    chown root:planner /etc/planner/planner.env
fi

echo "==> systemd unit files..."
cp /opt/planner/deploy/planner-api.service /etc/systemd/system/
cp /opt/planner/deploy/planner-bot.service /etc/systemd/system/
systemctl daemon-reload
echo "  units установлены; НЕ запускаем ещё (нужен planner.env и код)"

echo ""
echo "=== Bootstrap завершён ==="
echo "Следующий шаг: deploy/bin/deploy.sh"

**Создать `deploy/bin/deploy.sh`** — идемпотентный deploy (запускается на каждое обновление):

```bash
#!/usr/bin/env bash
# Deploy Личного Еженедельника на VPS.
# Запускается на каждое обновление сервера.
#
# Usage (на VPS как root):
#   sudo bash /opt/planner/deploy/bin/deploy.sh
#
# Что делает:
#   1. git pull
#   2. создать/обновить venv
#   3. pip install требований
#   4. alembic upgrade head (миграции)
#   5. рестарт systemd units
#   6. smoke test (curl /api/health)
set -euo pipefail

REPO_DIR=/opt/planner
VENV_DIR=$REPO_DIR/venv
ENV_FILE=/etc/planner/planner.env

echo "==> Проверка пререквизитов..."
if [ ! -f "$ENV_FILE" ]; then
    echo "ERROR: $ENV_FILE отсутствует. Запусти bootstrap-vps.sh и заполни env."
    exit 1
fi

if ! id planner &>/dev/null; then
    echo "ERROR: user 'planner' не существует. Запусти bootstrap-vps.sh"
    exit 1
fi

# Python version check
PY_VERSION=$(python3 --version 2>&1 | awk '{print $2}')
PY_MAJOR=$(echo "$PY_VERSION" | cut -d. -f1)
PY_MINOR=$(echo "$PY_VERSION" | cut -d. -f2)
if [ "$PY_MAJOR" -lt 3 ] || { [ "$PY_MAJOR" -eq 3 ] && [ "$PY_MINOR" -lt 10 ]; }; then
    echo "ERROR: Python >= 3.10 нужен, установлен $PY_VERSION"
    echo "  Установи: sudo apt install python3.12 python3.12-venv"
    exit 1
fi
echo "  Python $PY_VERSION — OK"

echo "==> git pull в $REPO_DIR..."
cd "$REPO_DIR"
sudo -u planner git fetch origin
sudo -u planner git reset --hard origin/main

echo "==> venv..."
if [ ! -d "$VENV_DIR" ]; then
    sudo -u planner python3 -m venv "$VENV_DIR"
fi

echo "==> pip install..."
sudo -u planner "$VENV_DIR/bin/pip" install --upgrade pip
sudo -u planner "$VENV_DIR/bin/pip" install -r "$REPO_DIR/server/requirements.txt"

echo "==> Alembic migrations..."
# Читаем DATABASE_URL из planner.env — он async (sqlite+aiosqlite), alembic нужен sync
# env.py сам конвертирует, но для передачи через CLI — export DATABASE_URL тоже
set -a
# shellcheck disable=SC1091
source "$ENV_FILE"
set +a

cd "$REPO_DIR/server"
sudo -u planner -E "$VENV_DIR/bin/alembic" upgrade head
cd "$REPO_DIR"

echo "==> Перезапуск systemd services..."
# Убедимся что unit files актуальны (могли измениться)
cp "$REPO_DIR/deploy/planner-api.service" /etc/systemd/system/
cp "$REPO_DIR/deploy/planner-bot.service" /etc/systemd/system/
systemctl daemon-reload

systemctl enable planner-api planner-bot
systemctl restart planner-api
systemctl restart planner-bot

echo "  Жду 3 секунды чтобы сервис поднялся..."
sleep 3

echo "==> Smoke test..."
# Локально (не через reverse proxy) — API на 127.0.0.1:8100
if curl -fsS http://127.0.0.1:8100/api/health; then
    echo ""
    echo "  /api/health OK"
else
    echo ""
    echo "  ERROR: /api/health не отвечает — проверь journalctl -u planner-api -n 50"
    exit 1
fi

echo ""
echo "==> systemctl status:"
systemctl status planner-api --no-pager -l | head -5
systemctl status planner-bot --no-pager -l | head -5

echo ""
echo "=== Deploy завершён ==="
echo "Публичный URL: https://planner.heyda.ru/api/health (через reverse-proxy — проверь отдельно)"
```

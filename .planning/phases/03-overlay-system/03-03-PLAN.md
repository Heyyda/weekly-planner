    ```python
    """
    Pillow-композитор иконки overlay/tray.

    Используется:
      - Plan 03-04 OverlayManager (статичная отрисовка 56x56 canvas image)
      - Plan 03-05 PulseAnimator (60fps перерисовка с pulse_t 0..1)
      - Plan 03-07 TrayManager (16/32px tray icon)

    Все цвета — verbatim из UI-SPEC §Color Palette + §Brand Identity.
    """
    from __future__ import annotations

    from PIL import Image, ImageDraw

    # ---- Цветовые константы (UI-SPEC verbatim) ----

    OVERLAY_BLUE_TOP    = (78, 161, 255)   # #4EA1FF (gradient top-left)
    OVERLAY_BLUE_BOTTOM = (30, 115, 232)   # #1E73E8 (gradient bottom-right)
    OVERLAY_RED_TOP     = (232, 90, 90)    # #E85A5A (overdue top)
    OVERLAY_RED_BOTTOM  = (192, 53, 53)    # #C03535 (overdue bottom)
    WHITE               = (255, 255, 255)
    BADGE_TEXT          = (30, 30, 30)     # тёмный текст badge

    # Размеры per UI-SPEC
    CORNER_RADIUS_FRAC  = 12 / 56          # 21.4% ≈ UI-SPEC (radius 12px на 56px)
    BADGE_SIZE_FRAC     = 16 / 56          # 28.6% (16px badge на 56px)
    ICON_SIZE_FRAC      = 0.55             # 60% → галочка/плюс

    # ---- Public API ----

    def render_overlay_image(
        size: int,
        state: str = "default",
        task_count: int = 0,
        overdue_count: int = 0,
        pulse_t: float = 0.0,
    ) -> Image.Image:
        """
        Создать финальную PIL.Image (RGBA) с rounded-square фоном + центральной иконкой + badge.

        size: пиксели (56 для overlay, 16/32 для tray)
        state: "default" | "empty" | "overdue"
        task_count: отображается в badge если state="default"
        overdue_count: отображается в badge если state="overdue"
        pulse_t: 0..1 — позиция в pulse-цикле (0=blue, 0.5=red, 1=blue).
                 Используется только для state="overdue".
        """
        # Валидация pulse_t (робастность)
        try:
            t = float(pulse_t)
        except (TypeError, ValueError):
            t = 0.0
        if t < 0.0:
            t = 0.0
        if t > 1.0:
            t = t - int(t)  # периодичность

        img = Image.new("RGBA", (size, size), (0, 0, 0, 0))
        draw = ImageDraw.Draw(img)

        # Цвет фона
        if state == "overdue":
            # Triangle-wave 0→1→0 к красному
            intensity = abs(t * 2.0 - 1.0)  # 0→1→0 за t 0..1 (wait: t=0.5 → intensity=0, t=0 → 1)
            # Исправим: t=0 → синий, t=0.5 → красный, t=1 → синий
            # intensity должна = 1 при t=0.5 (пик красного)
            intensity = 1.0 - abs(t * 2.0 - 1.0)
            bg_top    = _lerp_rgb(OVERLAY_BLUE_TOP,    OVERLAY_RED_TOP,    intensity)
            bg_bottom = _lerp_rgb(OVERLAY_BLUE_BOTTOM, OVERLAY_RED_BOTTOM, intensity)
        else:
            bg_top    = OVERLAY_BLUE_TOP
            bg_bottom = OVERLAY_BLUE_BOTTOM

        # Rounded square
        radius = max(2, int(size * CORNER_RADIUS_FRAC))
        if size >= 32:
            # Gradient для крупных иконок
            _draw_gradient_rounded(img, draw, bg_top, bg_bottom, radius)
        else:
            # Solid для tray 16px (gradient не читается)
            draw.rounded_rectangle(
                [(0, 0), (size - 1, size - 1)],
                radius=radius,
                fill=(*bg_bottom, 255),
            )

        # Центральная иконка (галочка или плюс)
        icon_size = int(size * ICON_SIZE_FRAC)
        ix = (size - icon_size) // 2
        iy = (size - icon_size) // 2
        if state == "empty":
            _draw_plus(draw, ix, iy, icon_size)
        else:
            _draw_checkmark(draw, ix, iy, icon_size)

        # Badge — только для крупных иконок
        badge_count = overdue_count if state == "overdue" else task_count
        if badge_count > 0 and size >= 32:
            _draw_badge(draw, size, badge_count)

        return img


    # ---- Internal helpers ----

    def _lerp_rgb(a: tuple, b: tuple, t: float) -> tuple:
        """Linear interpolation per RGB channel."""
        return tuple(int(a[i] + (b[i] - a[i]) * t) for i in range(3))


    def _draw_gradient_rounded(img: Image.Image, draw: ImageDraw.Draw,
                                top: tuple, bottom: tuple, radius: int) -> None:
        """Linear gradient top→bottom, затем clip под rounded rect через маску."""
        w, h = img.size
        # Заполняем gradient pixel-wise (для small 56x56 приемлемо)
        grad = Image.new("RGBA", (w, h), (0, 0, 0, 0))
        for y in range(h):
            t = y / max(1, h - 1)
            color = tuple(int(top[i] + (bottom[i] - top[i]) * t) for i in range(3))
            grad.paste(color + (255,), (0, y, w, y + 1))
        # Mask — rounded rect alpha
        mask = Image.new("L", (w, h), 0)
        mask_draw = ImageDraw.Draw(mask)
        mask_draw.rounded_rectangle([(0, 0), (w - 1, h - 1)], radius=radius, fill=255)
        img.paste(grad, (0, 0), mask)


    def _draw_checkmark(draw: ImageDraw.Draw, x: int, y: int, size: int) -> None:
        """Things 3 стиль — chunky галочка, stroke width ≈ size/7 (≈3px на 28px icon)."""
        w = max(3, size // 7)
        pts = [
            (x + int(size * 0.15), y + int(size * 0.55)),
            (x + int(size * 0.40), y + int(size * 0.75)),
            (x + int(size * 0.85), y + int(size * 0.25)),
        ]
        draw.line(pts, fill=(*WHITE, 255), width=w, joint="curve")


    def _draw_plus(draw: ImageDraw.Draw, x: int, y: int, size: int) -> None:
        """Плюс (empty state). Stroke width ≈ size/7 (3px на 28px icon)."""
        w = max(3, size // 7)
        cx, cy = x + size // 2, y + size // 2
        draw.line([(cx, y + int(size * 0.2)), (cx, y + int(size * 0.8))],
                  fill=(*WHITE, 255), width=w)
        draw.line([(x + int(size * 0.2), cy), (x + int(size * 0.8), cy)],
                  fill=(*WHITE, 255), width=w)


    def _draw_badge(draw: ImageDraw.Draw, size: int, count: int) -> None:
        """
        Белый ellipse 16x16 в правом верхнем углу с тёмным текстом count (max 99).

        Размер badge — 28.6% от size (BADGE_SIZE_FRAC), текст — размер 50% от badge.
        """
        bsize = max(8, int(size * BADGE_SIZE_FRAC))
        bx = size - bsize
        by = 0
        draw.ellipse([(bx, by), (bx + bsize, by + bsize)], fill=(*WHITE, 255))
        text = str(min(count, 99))
        # Используем дефолтный шрифт Pillow (не load_truetype — избежать проблем с путями в frozen exe)
        # Текст центрируется в ellipse
        try:
            bbox = draw.textbbox((0, 0), text)
            tw = bbox[2] - bbox[0]
            th = bbox[3] - bbox[1]
        except AttributeError:
            tw, th = bsize // 2, bsize // 2
        tx = bx + (bsize - tw) // 2
        ty = by + (bsize - th) // 2
        draw.text((tx, ty), text, fill=(*BADGE_TEXT, 255))
    ```

    """Unit-тесты icon_compose.render_overlay_image (Plan 03-03)."""
    import pytest
    from PIL import Image

    from client.ui.icon_compose import (
        render_overlay_image,
        OVERLAY_BLUE_BOTTOM, OVERLAY_RED_BOTTOM,
    )


    def test_returns_pil_image_correct_size():
        img = render_overlay_image(56, "default")
        assert isinstance(img, Image.Image)
        assert img.size == (56, 56)
        assert img.mode == "RGBA"


    def test_corner_pixel_transparent_rounded_clip():
        img = render_overlay_image(56, "default")
        # Угловые пиксели (0,0) и (55,55) — прозрачны из-за rounded corners
        r, g, b, a = img.getpixel((0, 0))
        assert a == 0, f"Corner (0,0) должен быть прозрачным, получили alpha={a}"


    def test_center_pixel_not_transparent():
        img = render_overlay_image(56, "default")
        r, g, b, a = img.getpixel((28, 28))
        assert a > 0, "Центр должен быть непрозрачным (галочка или фон)"


    def test_empty_state_has_plus_not_checkmark():
        img_empty = render_overlay_image(56, "empty")
        img_default = render_overlay_image(56, "default")
        # Empty и default — разные пиксели в районе центра (плюс vs галочка)
        # Горизонтальная черта плюса (y=28, x=15..45) — белая
        r, g, b, a = img_empty.getpixel((28, 28))
        assert (r, g, b) == (255, 255, 255) or a == 255  # центр креста — белый


    def test_badge_appears_when_count_positive_and_size_large():
        img = render_overlay_image(56, "default", task_count=3)
        # Badge location: правый верхний угол, ~(48, 8) внутри белого ellipse
        # Проверяем что (48, 8) — белый (WHITE = 255,255,255)
        r, g, b, a = img.getpixel((48, 8))
        assert (r, g, b) == (255, 255, 255), f"Badge area должен быть белым, got ({r},{g},{b})"


    def test_no_badge_when_count_zero():
        img = render_overlay_image(56, "default", task_count=0)
        # В месте badge должен быть цвет фона (синий), не белый
        r, g, b, a = img.getpixel((48, 8))
        # Не должно быть (255, 255, 255) — это был бы badge
        assert not (r == 255 and g == 255 and b == 255), "При task_count=0 badge быть не должно"


    def test_no_badge_on_small_icon():
        img = render_overlay_image(16, "default", task_count=5)
        # На 16x16 badge не отрисовывается (слишком мало)
        # Проверяем что в правом верхнем углу пиксель не белый
        r, g, b, a = img.getpixel((14, 1))
        assert not (r == 255 and g == 255 and b == 255)


    def test_overdue_pulse_zero_is_blue():
        img = render_overlay_image(56, "overdue", overdue_count=1, pulse_t=0.0)
        # Пиксель фона (не центр, не badge) — синий
        r, g, b, a = img.getpixel((10, 40))
        # Синий: B компонент >> R (для blue-dominant)
        assert b > r, f"pulse_t=0 должен быть синим (B>{r}>R): got RGB=({r},{g},{b})"


    def test_overdue_pulse_half_is_red():
        img = render_overlay_image(56, "overdue", overdue_count=1, pulse_t=0.5)
        # Пиксель фона — красный в пике pulse
        r, g, b, a = img.getpixel((10, 40))
        assert r > b, f"pulse_t=0.5 должен быть красным (R>{r}>B): got RGB=({r},{g},{b})"


    def test_overdue_badge_shows_overdue_count():
        img = render_overlay_image(56, "overdue", task_count=10, overdue_count=3)
        # Badge присутствует при overdue_count > 0
        r, g, b, a = img.getpixel((48, 8))
        assert (r, g, b) == (255, 255, 255), "Overdue badge должен быть белым"


    def test_robust_against_pulse_t_out_of_range():
        # Не должно крэшить при t > 1 или t < 0
        img = render_overlay_image(56, "overdue", overdue_count=1, pulse_t=1.5)
        assert img.size == (56, 56)
        img2 = render_overlay_image(56, "overdue", overdue_count=1, pulse_t=-0.3)
        assert img2.size == (56, 56)


    def test_tray_size_16_solid_fill():
        img = render_overlay_image(16, "default")
        assert img.size == (16, 16)

<objective>
PulseAnimator — 60fps driver который вызывает `overlay.refresh_image(state="overdue", pulse_t=t)`
каждые 16ms, где t — позиция в 2500ms цикле. Запускается когда хоть одна задача `is_overdue()`,
останавливается когда все сделаны/удалены.

Purpose: OVR-05 — критическая визуальная обратная связь пользователю. Pulse "зовёт внимание"
к просрочкам без навязчивого toast'а. Вынесено в отдельный модуль (не внутри OverlayManager)
потому что:
 1. Единственная ответственность = анимация (SRP)
 2. Проще тестировать в изоляции (mock overlay.refresh_image, check call args)
 3. Может быть заменён на pre-rendered frames (performance optimization) без изменения overlay

Критические pitfall'ы:
 - D-28 / PITFALL 2: ТОЛЬКО через root.after(), НЕ threading.Timer (Tk apartment)
 - Pulse cycle per UI-SPEC: 2.5s total = 1.0s blue → 0.75s red peak → 0.75s blue
 - Idempotent start/stop — guard против двойного запуска (overlap = 120fps chaos)

Output: `client/ui/pulse.py` PulseAnimator + unit tests.
</objective>

    """
    PulseAnimator — 60fps driver для overlay pulse animation (OVR-05).

    Цикл per UI-SPEC §Overdue:
        2500ms total
        t = 0.00..0.40  (1000ms): синий (плавный sin-подъём to peak в 0.5)
        t = 0.40..0.70  (750ms):  синий → красный → красный peak в t=0.5
        t = 0.70..1.00  (750ms):  красный → синий

    Упрощение: используем triangle wave (t → abs(2t-1) → 1 в 0.5, 0 в 0/1).
    В render_overlay_image pulse_t=0.5 = pик красного. Это match UI-SPEC.

    Критично: root.after() only (D-28). Никогда threading.Timer (PITFALL 2).
    """
    from __future__ import annotations

    import logging
    import time
    from typing import Callable, Optional

    logger = logging.getLogger(__name__)


    class PulseAnimator:
        """60fps pulse driver. Вызывает on_frame(pulse_t: float) каждые 16ms."""

        PULSE_INTERVAL_MS = 16     # ≈60fps per D-28
        PULSE_CYCLE_MS = 2500       # полный цикл 2.5s per UI-SPEC

        def __init__(self, root, on_frame: Callable[[float], None]) -> None:
            self._root = root
            self._on_frame = on_frame
            self._active: bool = False
            self._after_id: Optional[str] = None
            self._start_time_ms: int = 0

        def is_active(self) -> bool:
            return self._active

        def start(self) -> None:
            """Запустить pulse. Idempotent — повторный start no-op если уже активен."""
            if self._active:
                logger.debug("PulseAnimator.start: уже активен, игнорируем")
                return
            self._active = True
            self._start_time_ms = self._now_ms()
            self._tick()
            logger.debug("PulseAnimator started")

        def stop(self) -> None:
            """Остановить pulse + reset кадр на pulse_t=0.0 (blue)."""
            if not self._active:
                return
            self._active = False
            if self._after_id is not None:
                try:
                    self._root.after_cancel(self._after_id)
                except Exception as exc:
                    logger.debug("after_cancel failed: %s", exc)
                self._after_id = None
            # Reset frame — вернуть overlay в blue
            try:
                self._on_frame(0.0)
            except Exception as exc:
                logger.error("on_frame(reset) failed: %s", exc)
            logger.debug("PulseAnimator stopped")

        def _tick(self) -> None:
            """Один кадр анимации. Вычисляет pulse_t, вызывает on_frame, планирует next."""
            if not self._active:
                return
            elapsed = self._now_ms() - self._start_time_ms
            t = self._compute_pulse_t(elapsed)
            try:
                self._on_frame(t)
            except Exception as exc:
                logger.error("on_frame failed: %s", exc)
                self._active = False
                return
            # Планируем следующий кадр
            self._after_id = self._root.after(self.PULSE_INTERVAL_MS, self._tick)

        def _compute_pulse_t(self, elapsed_ms: int) -> float:
            """
            Позиция в pulse-цикле (0.0..1.0).

            elapsed_ms % PULSE_CYCLE_MS / PULSE_CYCLE_MS = текущая доля цикла.
            Интерпретация в icon_compose: t=0 blue, t=0.5 red peak, t=1 blue (wrap).
            """
            cycle_pos = elapsed_ms % self.PULSE_CYCLE_MS
            return cycle_pos / self.PULSE_CYCLE_MS

        @staticmethod
        def _now_ms() -> int:
            """Monotonic ms timestamp."""
            return int(time.monotonic() * 1000)
    ```

    """Unit-тесты PulseAnimator (Plan 03-05). Covers OVR-05."""
    from unittest.mock import MagicMock

    import pytest

    from client.ui.pulse import PulseAnimator


    def test_initial_state_inactive(headless_tk):
        cb = MagicMock()
        anim = PulseAnimator(headless_tk, on_frame=cb)
        assert not anim.is_active()
        assert cb.call_count == 0


    def test_start_activates(headless_tk):
        cb = MagicMock()
        anim = PulseAnimator(headless_tk, on_frame=cb)
        anim.start()
        assert anim.is_active()
        anim.stop()


    def test_start_idempotent(headless_tk):
        cb = MagicMock()
        anim = PulseAnimator(headless_tk, on_frame=cb)
        anim.start()
        first_after = anim._after_id
        anim.start()  # noop
        assert anim._after_id == first_after  # не перезаписан
        anim.stop()


    def test_stop_deactivates(headless_tk):
        cb = MagicMock()
        anim = PulseAnimator(headless_tk, on_frame=cb)
        anim.start()
        anim.stop()
        assert not anim.is_active()
        assert anim._after_id is None


    def test_stop_resets_to_blue(headless_tk):
        cb = MagicMock()
        anim = PulseAnimator(headless_tk, on_frame=cb)
        anim.start()
        cb.reset_mock()
        anim.stop()
        # stop() должен вызвать on_frame(0.0) для reset
        cb.assert_called_with(0.0)


    def test_tick_invokes_on_frame_at_least_once(headless_tk):
        cb = MagicMock()
        anim = PulseAnimator(headless_tk, on_frame=cb)
        anim.start()
        # Процессируем несколько tick'ов
        for _ in range(3):
            headless_tk.update()
            headless_tk.after(20)
            headless_tk.update_idletasks()
        anim.stop()
        assert cb.call_count >= 1


    def test_compute_pulse_t_zero_at_start(headless_tk):
        anim = PulseAnimator(headless_tk, on_frame=lambda t: None)
        assert anim._compute_pulse_t(0) == 0.0


    def test_compute_pulse_t_half_at_mid_cycle(headless_tk):
        anim = PulseAnimator(headless_tk, on_frame=lambda t: None)
        # 1250ms = PULSE_CYCLE_MS / 2 = peak red
        t = anim._compute_pulse_t(1250)
        assert abs(t - 0.5) < 0.01


    def test_compute_pulse_t_wraps_at_full_cycle(headless_tk):
        anim = PulseAnimator(headless_tk, on_frame=lambda t: None)
        assert anim._compute_pulse_t(2500) == 0.0


    def test_compute_pulse_t_wraps_beyond_cycle(headless_tk):
        anim = PulseAnimator(headless_tk, on_frame=lambda t: None)
        # 3000 % 2500 = 500; 500 / 2500 = 0.2
        t = anim._compute_pulse_t(3000)
        assert abs(t - 0.2) < 0.01


    def test_pulse_interval_is_16ms_60fps(headless_tk):
        assert PulseAnimator.PULSE_INTERVAL_MS == 16


    def test_pulse_cycle_is_2500ms(headless_tk):
        assert PulseAnimator.PULSE_CYCLE_MS == 2500


    def test_uses_root_after_not_threading_timer(headless_tk):
        """PITFALL 2 verification — grep уже проверяет, тут assert on usage."""
        import inspect
        source = inspect.getsource(PulseAnimator)
        assert "threading.Timer" not in source
        assert "root.after" in source

        cb = MagicMock()
        anim = PulseAnimator(headless_tk, on_frame=cb)
        anim.stop()  # без start — должно быть тихо
        cb.assert_not_called()
    ```

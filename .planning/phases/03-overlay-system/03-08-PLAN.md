    ```python
    """
    NotificationManager — 3 режима toast уведомлений (NOTIF-01..04).

    Режимы (UI-SPEC §Notifications):
      sound_pulse — Windows toast + overlay pulse + system sound (default)
      pulse_only  — overlay pulse only, no toast
      silent      — "Не беспокоить": ничего (pulse тоже не запускается через Plan 03-10)

    Pitfall-ы:
      PITFALL 3: winotify.Notification.show() блокирует thread (PowerShell subprocess).
                 Обязательно через threading.Thread(daemon=True).
      PITFALL 7: icon path — абсолютный. str(Path(path).resolve()).
    """
    from __future__ import annotations

    import logging
    import threading
    from datetime import datetime, timedelta, timezone
    from pathlib import Path
    from typing import Optional

    logger = logging.getLogger(__name__)

    VALID_MODES = {"sound_pulse", "pulse_only", "silent"}
    APPROACHING_WINDOW_MIN = 5  # за 5 минут до deadline
    OVERDUE_WINDOW_MIN = 1      # первая минута после deadline


    class NotificationManager:
        """Windows toast через winotify + deadline scheduler с dedup."""

        APP_ID = "Личный Еженедельник"

        def __init__(
            self,
            mode: str = "sound_pulse",
            icon_path: Optional[str] = None,
        ) -> None:
            self._mode: str = mode if mode in VALID_MODES else "sound_pulse"
            self._icon_path: Optional[str] = None
            if icon_path:
                self.set_icon(icon_path)
            # Dedup: (task_id, kind) → не отправлять повторно
            self._sent: set[tuple[str, str]] = set()

        # ---- Configuration ----

        def set_mode(self, mode: str) -> None:
            if mode in VALID_MODES:
                self._mode = mode
                logger.info("NotificationManager mode → %s", mode)
            else:
                logger.warning("Unknown notification mode: %r", mode)

        @property
        def mode(self) -> str:
            return self._mode

        def set_icon(self, icon_path: str) -> None:
            """PITFALL 7: всегда сохраняем АБСОЛЮТНЫЙ путь."""
            try:
                p = Path(icon_path).resolve()
                if p.exists():
                    self._icon_path = str(p)
                else:
                    logger.warning("Icon path не существует: %s", p)
                    self._icon_path = None
            except Exception as exc:
                logger.error("Icon path resolution failed: %s", exc)
                self._icon_path = None

        # ---- Toast API ----

        def send_toast(self, title: str, body: str) -> bool:
            """
            Отправить toast per текущий mode.

            Возвращает True если toast запущен (daemon thread), False — если заглушён.
            НЕ блокирует caller — winotify запускается в daemon thread (PITFALL 3).
            """
            if self._mode in ("silent", "pulse_only"):
                logger.debug("send_toast blocked (mode=%s)", self._mode)
                return False
            # PITFALL 3: daemon thread, winotify.show() блокирует
            thread = threading.Thread(
                target=self._do_show_toast,
                args=(title, body),
                daemon=True,
                name="winotify-toast",
            )
            thread.start()
            return True

        def _do_show_toast(self, title: str, body: str) -> None:
            try:
                from winotify import Notification
                kwargs = {
                    "app_id": self.APP_ID,
                    "title": title,
                    "msg": body,
                }
                if self._icon_path:
                    kwargs["icon"] = self._icon_path
                toast = Notification(**kwargs)
                toast.show()
                logger.info("Toast shown: %s", title)
            except Exception as exc:
                logger.error("winotify toast failed: %s", exc)

        # ---- Deadline scheduler ----

        def check_deadlines(self, tasks: list) -> list:
            """
            Найти задачи у которых приближается или только что прошёл deadline.

            Returns list of {"task_id", "title", "body", "kind"} для new-to-notify tasks.
            Dedup: уже уведомлённые (task_id + kind) пропускаются.
            """
            now = datetime.now(timezone.utc)
            results = []
            for task in tasks:
                deadline = self._parse_deadline(getattr(task, "time_deadline", None))
                if deadline is None:
                    continue
                if getattr(task, "done", False):
                    continue  # выполнено — не уведомлять
                if getattr(task, "deleted_at", None) is not None:
                    continue  # удалена

                delta_min = (deadline - now).total_seconds() / 60.0
                kind: Optional[str] = None
                if 0 <= delta_min <= APPROACHING_WINDOW_MIN:
                    kind = "approaching"
                elif -OVERDUE_WINDOW_MIN <= delta_min < 0:
                    kind = "overdue"

                if kind is None:
                    continue

                task_id = getattr(task, "id", None) or ""
                key = (task_id, kind)
                if key in self._sent:
                    continue
                self._sent.add(key)

                text = getattr(task, "text", "") or ""
                hm = deadline.astimezone().strftime("%H:%M")
                if kind == "approaching":
                    title = "Задача через 5 минут"
                    body = f"{text} — {hm}"
                else:
                    title = "Задача просрочена"
                    body = f"{text} — {hm}"
                results.append({
                    "task_id": task_id,
                    "title": title,
                    "body": body,
                    "kind": kind,
                })
            return results

        def fire_scheduled_toasts(self, tasks: list) -> None:
            """Синхронно дёрнуть check_deadlines + send_toast per match."""
            for item in self.check_deadlines(tasks):
                self.send_toast(item["title"], item["body"])

        # ---- Helpers ----

        @staticmethod
        def _parse_deadline(s: Optional[str]) -> Optional[datetime]:
            if not s:
                return None
            try:
                # Handle "...Z" suffix
                clean = s.replace("Z", "+00:00") if isinstance(s, str) else s
                dt = datetime.fromisoformat(clean)
                if dt.tzinfo is None:
                    dt = dt.replace(tzinfo=timezone.utc)
                return dt
            except (ValueError, TypeError):
                return None

        def reset_dedup(self) -> None:
            """Сбросить memory sent-уведомлений (вызывать при start-of-day или logout)."""
            self._sent.clear()
    ```

        text: str

    def test_default_mode_sound_pulse():
        n = NotificationManager()
        assert n.mode == "sound_pulse"


    def test_set_mode_valid():
        n = NotificationManager()
        n.set_mode("silent")
        assert n.mode == "silent"

        n = NotificationManager()
        n.set_mode("loud")

        n = NotificationManager(mode="silent")
        result = n.send_toast("hi", "body")
        assert result is False
        # Подождать возможный daemon thread (его не должно быть)
        time.sleep(0.05)
        assert len(mock_winotify) == 0


    def test_send_toast_pulse_only_blocks(mock_winotify):
        n = NotificationManager(mode="pulse_only")
        result = n.send_toast("hi", "body")
        assert result is False
        time.sleep(0.05)
        assert len(mock_winotify) == 0


    def test_send_toast_sound_pulse_fires(mock_winotify):
        n = NotificationManager(mode="sound_pulse")
        result = n.send_toast("hello", "body")
        assert result is True

        for _ in range(20):
            if mock_winotify:
                break
            time.sleep(0.05)
        assert len(mock_winotify) == 1

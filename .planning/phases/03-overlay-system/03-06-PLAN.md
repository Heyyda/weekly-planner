    ```python
    """
    MainWindow — главное окно планировщика (shell, Phase 3).

    Scope Phase 3:
      - Resizable CTkToplevel, min 320x320, default 460x600 (UI-SPEC)
      - Navigation header: ← стрелка, "Неделя N", → стрелка, "Сегодня" кнопка
      - Аккордеон placeholder-дней Пн-Вс (без содержимого — Phase 4 добавит)
      - Persistence window size+position через SettingsStore
      - Theme-aware через ThemeManager.subscribe

    Scope Phase 4:
      - Реальные task rendering
      - Навигация между неделями (работающая)
      - Add/edit/delete задач
      - Drag-and-drop
    """
    from __future__ import annotations

    import logging
    import tkinter as tk
    from datetime import date, timedelta
    from typing import Optional

    import customtkinter as ctk

    from client.ui.settings import SettingsStore, UISettings
    from client.ui.themes import FONTS, ThemeManager

    logger = logging.getLogger(__name__)

    DAY_NAMES_RU = ["Пн", "Вт", "Ср", "Чт", "Пт", "Сб", "Вс"]


    class MainWindow:
        """Главное окно — аккордеон дней + navigation. См. 03-UI-SPEC §Main Window."""

        MIN_SIZE = (320, 320)
        DEFAULT_SIZE = (460, 600)

        def __init__(
            self,
            root: ctk.CTk,
            settings_store: SettingsStore,
            settings: UISettings,
            theme_manager: ThemeManager,
        ) -> None:
            self._root = root
            self._settings_store = settings_store
            self._settings = settings
            self._theme = theme_manager

            self._window = ctk.CTkToplevel(root)
            self._window.withdraw()
            self._window.title("Личный Еженедельник")
            self._window.minsize(*self.MIN_SIZE)

            # Восстановить размер
            w, h = self._resolve_initial_size()
            self._window.geometry(f"{w}x{h}")

            # Восстановить позицию (если None — CTk автоцентрирует)
            pos = self._settings.window_position
            if pos is not None and len(pos) == 2:
                try:
                    x, y = int(pos[0]), int(pos[1])
                    self._window.geometry(f"{w}x{h}+{x}+{y}")
                except (TypeError, ValueError):
                    pass

            # Always-on-top из settings
            try:
                self._window.attributes("-topmost", self._settings.on_top)
            except tk.TclError:
                pass

            # Protocol — перехват close через tray, не реальный exit
            self._window.protocol("WM_DELETE_WINDOW", self._on_close)

            # Content
            self._day_sections: list[ctk.CTkFrame] = []
            self._build_ui()

            # Theme subscribe
            self._theme.subscribe(self._apply_theme)
            self._apply_theme({
                "bg_primary": self._theme.get("bg_primary"),
                "text_primary": self._theme.get("text_primary"),
                "bg_secondary": self._theme.get("bg_secondary"),
                "accent_brand": self._theme.get("accent_brand"),
            })

            # Сохранять размер при resize (через Configure event)
            self._window.bind("<Configure>", self._on_configure)

        # ---- Public API ----

        def show(self) -> None:
            self._window.deiconify()
            self._window.lift()

        def hide(self) -> None:
            self._window.withdraw()

        def toggle(self) -> None:
            """Wired to overlay.on_click — переключает visibility."""
            if self.is_visible():
                self.hide()
            else:
                self.show()

        def is_visible(self) -> bool:
            try:
                return self._window.winfo_viewable() != 0
            except tk.TclError:
                return False

        def set_always_on_top(self, enabled: bool) -> None:
            """OVR-06 propagation из OverlayManager.on_top_changed."""
            try:
                self._window.attributes("-topmost", enabled)
            except tk.TclError:
                pass

        def destroy(self) -> None:
            try:
                self._window.destroy()
            except Exception as exc:
                logger.debug("MainWindow destroy: %s", exc)

        # ---- Theme ----

        def _apply_theme(self, palette: dict) -> None:
            """ThemeManager.subscribe callback — перекрасить контейнеры."""
            bg = palette.get("bg_primary", "#F5EFE6")
            try:
                self._window.configure(fg_color=bg)
                if hasattr(self, "_root_frame"):
                    self._root_frame.configure(fg_color=bg)
                for section in self._day_sections:
                    section.configure(fg_color=palette.get("bg_secondary", bg))
            except tk.TclError:
                pass

        # ---- Layout ----

        def _build_ui(self) -> None:
            self._root_frame = ctk.CTkFrame(self._window, corner_radius=0)
            self._root_frame.pack(fill="both", expand=True)

            # Header — navigation placeholder
            header = ctk.CTkFrame(self._root_frame, height=40, corner_radius=0)
            header.pack(fill="x", side="top")
            ctk.CTkButton(header, text="←", width=32, command=lambda: None).pack(side="left", padx=4)
            ctk.CTkLabel(header, text="Неделя 16  14-20 апр",
                         font=FONTS["h1"]).pack(side="left", expand=True)
            ctk.CTkButton(header, text="Сегодня", width=72, command=lambda: None).pack(side="left", padx=4)
            ctk.CTkButton(header, text="→", width=32, command=lambda: None).pack(side="left", padx=4)

            # Scrollable content для аккордеона
            scroll = ctk.CTkScrollableFrame(self._root_frame)
            scroll.pack(fill="both", expand=True, padx=8, pady=4)

            # 7 placeholder day sections
            today = date.today()
            # weekday(): Mon=0..Sun=6
            monday = today - timedelta(days=today.weekday())
            for i in range(7):
                d = monday + timedelta(days=i)
                section = self._build_day_section(scroll, DAY_NAMES_RU[i], d, is_today=(d == today))
                self._day_sections.append(section)

        def _build_day_section(self, parent, day_name: str, d: date, is_today: bool) -> ctk.CTkFrame:
            """Placeholder-секция дня: заголовок + пустое тело. Phase 4 заменит."""
            section = ctk.CTkFrame(parent, corner_radius=6)
            section.pack(fill="x", pady=4)

            header_frame = ctk.CTkFrame(section, fg_color="transparent")
            header_frame.pack(fill="x")

            # Chevron + заголовок
            label_text = f"{day_name} {d.day}"
            if is_today:
                label_text += "  • сегодня"
            font = FONTS["body"]
            if is_today:
                font = (font[0], font[1], "bold")
            ctk.CTkLabel(header_frame, text=label_text, font=font).pack(side="left", padx=8)
            # Placeholder счётчика "(0)"
            ctk.CTkLabel(header_frame, text="(0)", font=FONTS["caption"]).pack(side="right", padx=8)
            return section

        # ---- Persistence ----

        def _resolve_initial_size(self) -> tuple:
            ws = self._settings.window_size
            try:
                w, h = int(ws[0]), int(ws[1])
                if w >= self.MIN_SIZE[0] and h >= self.MIN_SIZE[1]:
                    return (w, h)
            except (TypeError, ValueError, IndexError):
                pass
            return self.DEFAULT_SIZE

        def _on_configure(self, event) -> None:
            """Resize event — сохранить если размер изменился."""
            # Фильтр: только события именно окна, не child widgets
            if event.widget is self._window:
                try:
                    new_size = [self._window.winfo_width(), self._window.winfo_height()]
                    new_pos = [self._window.winfo_x(), self._window.winfo_y()]
                    # Обновим settings только если действительно отличается
                    if new_size != self._settings.window_size:
                        self._settings.window_size = new_size
                        self._settings.window_position = new_pos
                        # Не сохраняем на каждый пиксель — defer до _on_close
                except tk.TclError:
                    pass

        def _on_close(self) -> None:
            """WM_DELETE → сохранить состояние и hide (не destroy). App control exit через tray."""
            self._save_window_state()
            self.hide()

        def _save_window_state(self) -> None:
            try:
                self._settings.window_size = [self._window.winfo_width(),
                                              self._window.winfo_height()]
                self._settings.window_position = [self._window.winfo_x(),
                                                  self._window.winfo_y()]
                self._settings_store.save(self._settings)
                logger.debug("MainWindow state saved: %s", self._settings.window_size)
            except tk.TclError as exc:
                logger.debug("_save_window_state skip: %s", exc)
    ```

        assert mw._window.winfo_exists()

    def test_initially_hidden(mw_deps):
        mw = MainWindow(mw_deps["root"], mw_deps["settings_store"],
                        mw_deps["settings"], mw_deps["theme"])
        mw_deps["root"].update()
        assert not mw.is_visible()

                        mw_deps["settings"], mw_deps["theme"])

        mw_deps["root"].update()

        app._setup()
        yield app
        try:
            app._handle_quit()
        except Exception:
            pass


    def test_e2e_setup_all_components_ready(authed_app):
        assert authed_app.storage is not None
        assert authed_app.sync is not None
        assert authed_app.overlay is not None
        assert authed_app.main_window is not None
        assert authed_app.pulse is not None
        assert authed_app.notifications is not None
        assert authed_app.tray is not None
        assert authed_app.theme is not None


    def test_overlay_click_toggles_main_window(authed_app):
        """OVR-04: single-click по overlay → main window open/close."""
        # main window начинает hidden
        assert authed_app.main_window.is_visible() is False
        # Симулируем overlay.on_click()
        authed_app.overlay.on_click()
        authed_app.root.update()
        assert authed_app.main_window.is_visible() is True
        authed_app.overlay.on_click()
        authed_app.root.update()
        assert authed_app.main_window.is_visible() is False


    def test_overdue_task_triggers_pulse(authed_app):
        """OVR-05: overdue задача → pulse автоматически активируется."""
        from client.core.models import Task
        yesterday = (date.today() - timedelta(days=1)).isoformat()
        task = Task.new(user_id="u1", text="Old task", day=yesterday)
        authed_app.storage.add_task(task)
        authed_app._refresh_ui()
        assert authed_app.pulse.is_active() is True


    def test_done_overdue_stops_pulse(authed_app):
        """OVR-05: done задача → pulse останавливается."""
        from client.core.models import Task
        yesterday = (date.today() - timedelta(days=1)).isoformat()
        task = Task.new(user_id="u1", text="Old task", day=yesterday)
        authed_app.storage.add_task(task)
        authed_app._refresh_ui()
        assert authed_app.pulse.is_active() is True
        # Отметить выполненной
        authed_app.storage.update_task(task.id, done=True)
        authed_app._refresh_ui()
        assert authed_app.pulse.is_active() is False


    def test_theme_switch_persists_to_settings(authed_app):
        """TRAY-03: смена темы через tray сохраняется в settings.json мгновенно."""
        authed_app.tray._cb_theme("dark")
        authed_app.root.update_idletasks()
        assert authed_app.settings.theme == "dark"
        # Проверим что сохранено на disk
        reloaded = authed_app.settings_store.load()
        assert reloaded.theme == "dark"


    def test_on_top_toggle_propagates_to_main_window(authed_app):
        """OVR-06: toggle через tray применяется и к overlay и к main window."""
        authed_app.settings.on_top = True
        authed_app.tray._cb_toggle_on_top()
        authed_app.root.update_idletasks()
        assert authed_app.settings.on_top is False
        # Main window topmost тоже False (проверяется через attributes, может быть string)


    def test_rapid_20_tray_callbacks_no_crash(authed_app):
        """TRAY-04: 20 quick callbacks → no RuntimeError: main thread is not in main loop."""
        for _ in range(20):
            authed_app.tray._cb_show()
            authed_app.root.update_idletasks()
        # Нет exception — success. Также main_window должен в visible state (последний show)
        assert authed_app.main_window.is_visible() is True


    def test_notifications_silent_blocks_toast(authed_app, mock_winotify):
        """NOTIF-04: mode=silent → fire_scheduled_toasts не создаёт toast."""
        from client.core.models import Task, utcnow_iso
        from datetime import datetime, timezone
        # Включаем silent
        authed_app.notifications.set_mode("silent")
        # Создаём задачу с deadline через 3 мин
        soon = (datetime.now(timezone.utc) + timedelta(minutes=3))
        deadline_iso = soon.isoformat().replace("+00:00", "Z")
        task = Task.new(user_id="u1", text="Deadline task", day=date.today().isoformat(),
                        time_deadline=deadline_iso)
        authed_app.storage.add_task(task)
        authed_app._scheduled_deadline_check()
        time.sleep(0.15)
        assert len(mock_winotify) == 0


    def test_notifications_sound_pulse_fires_toast(authed_app, mock_winotify):
        """NOTIF-01 + NOTIF-02 + NOTIF-03: sound_pulse mode + approaching deadline → toast."""
        from client.core.models import Task
        from datetime import datetime, timezone
        authed_app.notifications.set_mode("sound_pulse")
        soon = (datetime.now(timezone.utc) + timedelta(minutes=3))
        deadline_iso = soon.isoformat().replace("+00:00", "Z")
        task = Task.new(user_id="u1", text="Deadline task",
                        day=date.today().isoformat(), time_deadline=deadline_iso)
        authed_app.storage.add_task(task)
        authed_app._scheduled_deadline_check()
        # Ждём daemon thread
        for _ in range(20):
            if mock_winotify:
                break
            time.sleep(0.05)
        assert len(mock_winotify) >= 1


    def test_autostart_toggle_roundtrip(authed_app, mock_winreg):
        """Tray toggle autostart → winreg updated."""
        import winreg
        from client.utils.autostart import APP_REG_VALUE, REG_PATH

        # Изначально пусто
        key = (winreg.HKEY_CURRENT_USER, REG_PATH, APP_REG_VALUE)
        assert key not in mock_winreg
        # Toggle ON
        authed_app._handle_autostart_changed(True)
        assert key in mock_winreg
        # Toggle OFF
        authed_app._handle_autostart_changed(False)
        assert key not in mock_winreg


    def test_quit_stops_all_subsystems(authed_app):
        """_handle_quit: pulse, sync, tray, overlay, main_window — все closed."""
        # Start pulse to check stop
        authed_app.pulse.start()
        assert authed_app.pulse.is_active() is True

        authed_app._handle_quit()
        # После quit
        assert authed_app._quit_requested is True
        assert authed_app.pulse.is_active() is False
        # mock_pystray icon.stopped
        fake_icons = authed_app.tray._icon
        if fake_icons is not None:
            # Fake icon stopped
            assert getattr(fake_icons, "stopped", False) is True


    def test_logout_stops_sync_and_clears_auth(authed_app):
        """Разлогиниться → auth.logout + sync.stop."""
        logout_calls = []
        original_logout = authed_app.auth.logout
        authed_app.auth.logout = lambda: logout_calls.append(1)
        authed_app._handle_logout()
        assert logout_calls == [1]
        assert authed_app._authenticated is False

    # ---------- Phase 4 UI fixtures ----------

    @pytest.fixture
    def timestamped_task_factory():
        """Фабрика Task'ов с настраиваемыми day/time/done для UI-тестов.

        Usage:
            def test_overdue(timestamped_task_factory):
                task = timestamped_task_factory(text="old", day_offset=-1)
                assert task.is_overdue() is True

            def test_with_time(timestamped_task_factory):
                task = timestamped_task_factory(day_offset=0, time="14:30")
                assert task.time_deadline == "14:30"
        """
        from datetime import date, timedelta
        from client.core.models import Task

        def factory(
            text: str = "Test task",
            day_offset: int = 0,
            time: str | None = None,
            done: bool = False,
            user_id: str = "test-user",
            position: int = 0,
        ) -> Task:
            target_date = (date.today() + timedelta(days=day_offset)).isoformat()
            task = Task.new(
                user_id=user_id,
                text=text,
                day=target_date,
                time_deadline=time,
                position=position,
            )
            task.done = done
            return task

        return factory


    @pytest.fixture
